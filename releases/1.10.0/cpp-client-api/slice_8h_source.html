291 inline int Slice::compare(const Slice& b) const {
292  const int min_len = (size_ < b.size_) ? size_ : b.size_;
293  int r = MemCompare(data_, b.data_, min_len);
294  if (r == 0) {
295  if (size_ < b.size_) r = -1;
296  else if (size_ > b.size_) r = +1;
297  }
298  return r;
299 }
