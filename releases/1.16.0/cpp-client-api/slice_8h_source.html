336 inline int Slice::compare(const Slice& b) const {
337  const size_t min_len = (size_ < b.size_) ? size_ : b.size_;
338  const int r = MemCompare(data_, b.data_, min_len);
339  if (r != 0) {
340  return r;
341  }
342  if (size_ < b.size_) {
343  return -1;
344  }
345  if (size_ > b.size_) {
346  return 1;
347  }
348  return 0;
349 }
