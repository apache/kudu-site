292 inline int Slice::compare(const Slice& b) const {
293  const int min_len = (size_ < b.size_) ? size_ : b.size_;
294  int r = MemCompare(data_, b.data_, min_len);
295  if (r == 0) {
296  if (size_ < b.size_) r = -1;
297  else if (size_ > b.size_) r = +1;
298  }
299  return r;
300 }
